1function array = dpqs(array)
  2	%DPQS Implements a dual pivot quick sort
  3	
  4	array = DPQSWorker(array,1,numel(array));
  5	
  6end

  7
  8function array = DPQSWorker(array, lowerBound, upperBound)
  9	
 10	if(upperBound <= lowerBound)
 11		return;
 12	end
 13	
 14	pivot1Index = lowerBound;
 15	pivot2Index = upperBound;
 16
 17	prevIndex1 = pivot2Index;
 18	prevIndex2 = pivot1Index;
 19	
 20	run = true;
 21	
 22	while(run)
 23		
 24		run = false;
 25		
 26		for i = prevIndex1:-1:pivot1Index+1
 27			
 28			if(array(i) < array(pivot1Index))
 29				if(i ~= pivot2Index)
 30					while(array(pivot1Index+1) < array(pivot1Index))
 31						tmp = array(pivot1Index);
 32						array(pivot1Index) = array(pivot1Index+1);
 33						array(pivot1Index+1) = tmp;
 34						pivot1Index = pivot1Index+1;
 35						if(pivot1Index > prevIndex2)
 36							prevIndex2 = pivot1Index;
 37						end
 38					end
 39					
 40					if(i <= pivot1Index)
 41						break;
 42					end
 43					
 44					tmp = array(pivot1Index);
 45					array(pivot1Index) = array(i);
 46					array(i) = array(pivot1Index+1);
 47					array(pivot1Index+1) = tmp;
 48					pivot1Index = pivot1Index + 1;
 49					prevIndex1 = i;
 50				else
 51					tmp = array(pivot1Index);
 52					array(pivot1Index) = array(pivot2Index);
 53					array(pivot2Index) = tmp;
 54				end
 55				
 56				run = true;
 57				
 58				break;
 59				
 60			end
 61			
 62		end
 63			
 64			for i = prevIndex2:pivot2Index-1
 65				
 66				if(array(i) > array(pivot2Index))
 67					
 68					if(i ~= pivot1Index)
 69						
 70						while(array(pivot2Index-1) > array(pivot2Index))
 71							
 72							tmp = array(pivot2Index);
 73							array(pivot2Index) = array(pivot2Index-1);
 74							array(pivot2Index-1) = tmp;
 75							pivot2Index = pivot2Index - 1;
 76							if(pivot2Index < prevIndex1)
 77								prevIndex1 = pivot2Index;
 78							end
 79						end
 80						
 81						if(i >= pivot2Index)
 82							break;
 83						end
 84						
 85						tmp = array(pivot2Index);
 86						array(pivot2Index) = array(i);
 87						array(i) = array(pivot2Index-1);
 88						array(pivot2Index-1) = tmp;
 89						pivot2Index = pivot2Index -1;
 90						prevIndex2 = i;
 91					else
 92						tmp = array(pivot2Index);
 93						array(pivot2Index) = array(pivot1Index);
 94						array(pivot1Index) = tmp;
 95					end
 96					run = true;
 97					
 98					break;
 99					
100				end
101			end
102			
103	end
104		
105		array = DPQSWorker(array, lowerBound, pivot1Index - 1);
106		array = DPQSWorker(array, pivot1Index + 1, pivot2Index - 1);
107		array = DPQSWorker(array, pivot2Index + 1, upperBound);
108		
109end
